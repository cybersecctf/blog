def msb(num, pos):
    ret = (num & (1 << pos[0])) >> pos[0]
    for p in pos[1:]:
        ret ^= (num & (1 << p)) >> p
    return ret

def get_next(last, pos):
    m = msb(last, pos)
    return (m << 7)|(last >> 1)

def solve(cipher,prefix,msb_pos):
   
    last = 0

        output += chr(cipher[i] ^ last)
